local Players = game:GetService("Players")
local MarketplaceService = game:GetService("MarketplaceService")
local player = Players.LocalPlayer
local date = os.date("%d/%m/%Y")
local gameId = game.PlaceId

local success, info = pcall(function()
    return MarketplaceService:GetProductInfo(gameId)
end)

local gameName = success and info.Name or "Unknown Game"

WindUI:Notify({
    Title = "Thông tin sử dụng",
    Content = string.format(
        "Tên player: %s\nNgày dùng: %s\nGame: %s\nGameId: %s",
        player.Name, date, gameName, tostring(gameId)
    ),
    Duration = 8,
    Icon = "info"
})
